106 задача. Преобразуйте строку 'var_test_text' в 'varTestText'. Скрипт, конечно же, должен работать с любыми аналогичными строками. 

        // превращает нижнее подчеркивание в верблюжий способ 
        function underscoreToCamelcase( str ) {
            const arr = str.split('_');
            // ['var' 'test', 'text']

            for (let i = 1; i < arr.length; i++) {
                arr[i] = arr[i][0].toUpperCase() + arr[i].slice(1);
            }
            // ['var', 'Test', 'Text']

            const new_str = arr.join(''); // объединить массив в строку через ;
            return new_str;
        } 

        let example1 = "var_test_text";
        alert(`${example1} -> ${underscoreToCamelcase(example1)}`);
